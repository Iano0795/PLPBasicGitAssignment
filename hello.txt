"""
This code snippet demonstrates the usage of a Python function to calculate the sum of two numbers.
"""

def add_numbers(a, b):
    """
    Adds two numbers and returns the sum.

    Parameters:
    a (int): The first number.
    b (int): The second number.

    Returns:
    int: The sum of the two numbers.
    """
    return a + b
